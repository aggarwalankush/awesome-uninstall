# MAC

```bash
# Jetbrains config
sudo rm -rf ~/Library/Saved\ Application\ State/com.jetbrains.*
sudo rm -rf ~/Library/Preferences/com.jetbrains.*
sudo rm -rf ~/Library/Preferences/jetbrains.*
sudo rm -rf ~/Library/Caches/com.jetbrains.*
sudo rm -rf ~/Library/Caches/jetbrains.*
sudo rm -rf ~/Library/Application\ Support/com.apple.sharedfilelist/com.apple.LSSharedFileList.ApplicationRecentDocuments/com.jetbrains.*

# IntelliJ IDEA Ultimate Edition
sudo rm -rf ~/Library/Preferences/IntelliJIdea*
sudo rm -rf ~/Library/Caches/IntelliJIdea*
sudo rm -rf ~/Library/Application\ Support/IntelliJIdea*
sudo rm -rf ~/Library/Logs/IntelliJIdea*

# IntelliJ IDEA Community Edition
sudo rm -rf /usr/local/bin/idea
sudo rm -rf ~/Library/Preferences/IdeaIC*
sudo rm -rf ~/Library/Caches/IdeaIC*
sudo rm -rf ~/Library/Application\ Support/IdeaIC*
sudo rm -rf ~/Library/Logs/IdeaIC*
```
